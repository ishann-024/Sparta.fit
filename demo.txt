@PostMapping("/fill-assessment-form")
    public ResponseEntity<String> submitAssessment(@RequestBody AssessmentFormDTO assessmentForm) {
        try {
            teamLeadService.submitAssessment(assessmentForm);
            return ResponseEntity.ok("Assessment submitted successfully");
        } catch (IllegalArgumentException e) {
            return ResponseEntity.badRequest().body(e.getMessage());
        } catch (Exception e) {
            return ResponseEntity.internalServerError().body("Error submitting assessment: " + e.getMessage());
        }
    }

public class AssessmentFormDTO {
    private Integer interviewId;
    private InterviewStatus status; 
    private String remarks;
    private Double score;
    private Integer candidateId;
    private Integer jobRequestId;
    private Integer roundNumber;
}

public enum InterviewStatus {
	PENDING,INTERVIEW1_CLEARED,INTERVIEW2_CLEARED,HR_CLEARED,
    INTERVIEW1_REJECTED,INTERVIEW2_REJECTED,HR_REJECTED;
    
 // Convert String to InterviewStatus
    public static InterviewStatus fromString(String status) {
        if (status == null) {
            return null;
        }
        try {
            return InterviewStatus.valueOf(status.toUpperCase());
        } catch (IllegalArgumentException e) {
            throw new IllegalArgumentException("Invalid interview status: " + status);
        }
    }

    // Convert InterviewStatus to String for database
    public String getDbValue() {
        return this.name();
    }

    // Check if status indicates clearing current round
    public boolean isCleared() {
        return this == INTERVIEW1_CLEARED || this == INTERVIEW2_CLEARED || this == HR_CLEARED;
    }

    // Check if status indicates rejection
    public boolean isRejected() {
        return this == INTERVIEW1_REJECTED || this == INTERVIEW2_REJECTED || this == HR_REJECTED;
    }
}
